    User {
        int Id PK
        string UserName
        string Email
        text PasswordHash
    }

    RefreshToken {
        int Id PK
        string Token
        int UserId FK
        datetime CreatedAt
        datetime ExpiresAt
        datetime RevokedAt
    }

    BaseProduct {
        int Id PK
        string Name
        bool IsDeleted
        decimal Calories
        decimal Protein
        decimal Fat
        decimal Carbs
    }

    UserProduct {
        int Id PK
        int UserId FK
        int BaseProductId FK
        bool IsDeleted
        decimal CustomCalories
        decimal CustomProtein
        decimal CustomFat
        decimal CustomCarbs
    }

    UserDish {
        int Id PK
        int UserId FK
        string Name
        bool IsReadyDish
        bool IsDeleted
        decimal Calories
        decimal Protein
        decimal Fat
        decimal Carbs
    }

    UserDishIngredient {
        int Id PK
        int UserDishId FK
        int UserProductId FK
        double Quantity
        bool IsDeleted
        decimal Calories
        decimal Protein
        decimal Fat
        decimal Carbs
    }

    UserMealEntry {
        int Id PK
        int UserId FK
        int UserDishId FK
        date Date
        decimal TotalQuantity
        bool IsDeleted
        datetime CreatedAt
        decimal TotalCalories
        decimal TotalProtein
        decimal TotalFat
        decimal TotalCarbs
    }

    UserMealEntryIngredient {
        int Id PK
        int UserMealEntryId FK
        int UserProductId FK
        decimal Quantity
        string ProductNameSnapshot
        bool IsDeleted
        decimal CaloriesSnapshot
        decimal ProteinSnapshot
        decimal FatSnapshot
        decimal CarbsSnapshot
    }
